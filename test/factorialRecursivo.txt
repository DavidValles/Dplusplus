func integer factorial(integer n) {
    if (n < 1) {
        return -1;
    }
    if (n == 1) {
        return 1;
    }
    var integer n1 = factorial(n - 1);
    return n * n1;
}

main {
    var integer input;

    print("Enter factorial number from 1 - 12");
    read(input);
    while (input != -1) {
        if (input < 1 or input > 12) {
            print("Incorrect range");
        }
        else {
            print("Answer ", factorial(input));
        }
        read(input);
    }
}
